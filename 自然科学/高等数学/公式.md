// ============================================
// Obsidian DataviewJS - 数学公式标签显示脚本 v2.0
// 功能：显示 #数学/类别/公式 及其子标签的内容
// 支持格式：行内格式、列表格式
// ============================================

const baseTag = "#数学/公式";
const baseTagPattern = baseTag.slice(1); // 移除 # 号

// 存储解析后的数据
const tagData = {};

// 获取子标签名称（取第一级子标签）
function getSubTagName(tag) {
    const prefix = baseTagPattern + "/";
    const idx = tag.indexOf(prefix);
    if (idx !== -1) {
        const rest = tag.slice(idx + prefix.length);
        return rest.split("/")[0];
    }
    return null;
}

// 获取所有包含标签的页面
const pages = dv.pages(baseTag);

for (const page of pages) {
    const file = app.vault.getAbstractFileByPath(page.file.path);
    if (!file) continue;

    const content = await app.vault.read(file);
    const lines = content.split("\n");

    let currentSubTag = null;
    let listItems = [];
    let lastTagLine = -1;

    for (let i = 0; i < lines.length; i++) {
        const line = lines[i];
        const trimmedLine = line.trim();

        // 跳过空行
        if (trimmedLine === "") {
            continue;
        }

        // 匹配行内格式：#数学/类别/公式/xxx 内容
        const inlineRegex = new RegExp(`#${baseTagPattern}/([^\\s#]+)\\s+(.+)`);
        const inlineMatch = trimmedLine.match(inlineRegex);

        if (inlineMatch) {
            // 先保存之前累积的列表
            if (currentSubTag && listItems.length > 0) {
                if (!tagData[currentSubTag]) tagData[currentSubTag] = [];
                tagData[currentSubTag].push({
                    type: "list",
                    items: [...listItems],
                    source: page.file.link
                });
                listItems = [];
            }

            const subTag = inlineMatch[1].split("/")[0];
            const contentText = inlineMatch[2].trim();

            if (!tagData[subTag]) tagData[subTag] = [];
            tagData[subTag].push({
                type: "inline",
                content: contentText,
                source: page.file.link
            });

            currentSubTag = subTag;
            lastTagLine = i;
            continue;
        }

        // 匹配标签单独一行（后面是列表）
        const tagOnlyRegex = new RegExp(`^#${baseTagPattern}(/[^\\s#]+)?$`);
        const tagOnlyMatch = trimmedLine.match(tagOnlyRegex);

        if (tagOnlyMatch) {
            // 保存之前的列表
            if (currentSubTag && listItems.length > 0) {
                if (!tagData[currentSubTag]) tagData[currentSubTag] = [];
                tagData[currentSubTag].push({
                    type: "list",
                    items: [...listItems],
                    source: page.file.link
                });
                listItems = [];
            }

            // 确定子标签
            if (tagOnlyMatch[1]) {
                currentSubTag = tagOnlyMatch[1].slice(1).split("/")[0];
            } else {
                currentSubTag = "_default_";
            }

            lastTagLine = i;
            continue;
        }

        // 匹配列表项（必须紧跟在标签行之后）
        const listMatch = trimmedLine.match(/^- (.+)/);
        if (listMatch && currentSubTag) {
            listItems.push(listMatch[1].trim());
            lastTagLine = i;
            continue;
        }

        // 遇到非列表内容，保存当前列表并重置
        if (listItems.length > 0 && currentSubTag) {
            if (!tagData[currentSubTag]) tagData[currentSubTag] = [];
            tagData[currentSubTag].push({
                type: "list",
                items: [...listItems],
                source: page.file.link
            });
            listItems = [];
        }

        // 非列表行重置当前标签
        if (!trimmedLine.startsWith("-")) {
            currentSubTag = null;
        }
    }

    // 处理文件末尾的列表
    if (listItems.length > 0 && currentSubTag) {
        if (!tagData[currentSubTag]) tagData[currentSubTag] = [];
        tagData[currentSubTag].push({
            type: "list",
            items: [...listItems],
            source: page.file.link
        });
    }
}

// ============================================
// 渲染输出
// ============================================

const subTags = Object.keys(tagData).sort((a, b) => {
    // 默认分类排在最后
    if (a === "_default_") return 1;
    if (b === "_default_") return -1;
    return a.localeCompare(b, "zh-CN");
});

if (subTags.length === 0) {
    dv.paragraph("⚠️ 未找到相关内容");
} else {
    // 统计信息
    const totalItems = Object.values(tagData).reduce((sum, arr) => sum + arr.length, 0);
    dv.paragraph(`📊 共找到 **${subTags.length}** 个分类，**${totalItems}** 条内容`);

    dv.paragraph("---");

    for (const subTag of subTags) {
        const items = tagData[subTag];

        // 显示分类标题
        if (subTag === "_default_") {
            dv.header(3, "📋 公式（未分类）");
        } else {
            dv.header(3, `📐 ${subTag}`);
        }

        // 渲染内容
        for (const item of items) {
            if (item.type === "inline") {
                // 行内格式
                dv.paragraph(`${item.content}`);
            } else if (item.type === "list") {
                // 列表格式
                const listContent = item.items.map(i => `- ${i}`).join("\n");
                dv.paragraph(listContent);
            }
        }

        dv.paragraph("");
    }
}
