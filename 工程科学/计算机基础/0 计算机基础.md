```dataviewjs
// 配置区：使用时修改这两个变量即可
const targetFolder = "#计算机/软件应用/快捷键"; // 要搜索的文件夹
//const searchTerm = "快捷键"; // 要搜索的关键词

// 获取目标文件夹下所有页面
const pages = dv.pages(targetFolder);
const filePaths = pages.map(p => p.file.path);

// 获取所有匹配的Markdown文件
const files = app.vault.getMarkdownFiles()
    .filter(f => filePaths.includes(f.path));

// 异步处理每个文件
const processFiles = files.map(async (file) => {
    const content = await app.vault.cachedRead(file);
    const lines = content.split("\n")
        .filter(line => line.includes(targetFolder)) 
        .map(line => line.replace(/^[>\s\.\-0-9]+|#+.*$/g, '').trim());
        
    // 获取美观的文件链接
    const link = pages.find(p => p.file.path === file.path)?.file.link;
    
    return [
        link || file.name, // 防错处理
        lines.filter(l => l) // 移除空行
    ];
});

// 展示结果
Promise.all(processFiles)
    .then(results => {
        // 过滤掉没有匹配结果的文件
        const nonEmptyResults = results.filter(([_, lines]) => lines.length > 0);
        dv.table(["文件", "快捷键"], nonEmptyResults);
    });
```
