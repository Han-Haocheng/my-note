## 操作系统运转流程

### 操作系统启动流程(X86)

1. CPU处于实模式
	- 实模式：寻址方式为`CS<<4:IP`
2. `CS=0xFFFF;IP=0x0000`
	- 段寄存器(CS)：存储当前执行指令所在代码段的基地址
	- 偏移寄存器(IP)：指定内存地址的在代码端基地址的偏移量
3. 寻址ROM中的BIOS映射区起始地址
	- ROM BIOS映射区：`0xFFFF0`
4. 检查RAM、键盘、显示器、软硬磁盘等硬件设备
5. 将磁盘0磁道0扇区写入`0x7C00`
	- 系统引导扇区(0磁道0扇区)：启动设备的第一个扇区
6. 设置`CS=0x07C0;IP=0x0000`

### 引导扇区代码

引导扇区代码：bootsect.s文件

#### 数据定义


```arm-asm

.globl begtext, begdata, begbss, endtext, enddata, endbss
.text
begtext:
.data
begdata:
.bss
begbss:
.text

SYSSIZE = 0x8000        ; 该变量可根据Image大小设定
SETUPLEN= 4             ; setup模块的长度，4个扇区
BOOTSEG = 0x07c0        ; 启动扇区的起始地址
INITSEG = 0x9000        ; 引导程序把自身搬运到0x90000
SETUPSEG= 0x9020        ; setup模块被加载到0x90200
SYSSEG  = 0x1000        ; system模块被加载到0x10000
ENDSEG  = SYSSEG+SYSSIZE; 停止加载地址

```


#### 搬运启动程序

```
entry  start            ; entry确定程序入口
start:                  ; 使用寄存 ax ds:si es:di
    mov ax,#BOOTSEG
    mov ds,ax           ; 将ds设置为BOOTSEG=0x07C0
    sub si,si           ; 将ds段偏移设置为0
    mov ax,#INITSEG
    mov es,ax           ; 将es设置为INITSEG=0x9000
    sub di,di           ; 将es段偏移设置为0
    
    mov cx,#0x100       ; 准备循环256次
    rep movw            ; 每次搬运2字节
    
    jmpi go,INITSEG     ; 间接跳转跳转到INITSEG:go位置
    
```

- 启动程序将自身从`0x07C00`搬运到`0x90000`
- 然后跳转到`0x9000:go`位置
- 代码段：`0x90000-0x90200`
 

| 内存范围            | 内容    |
| --------------- | ----- |
| 0x07c00         | 旧启动程序 |
| 0x90000-0x90100 | 新启动程序 |

#### 加载setup模块

```
go: mov ax,cs          
    mov ds,ax            
    mov es,ax            
    mov ss,ax      ; cs=ds=es=ss=0x9000
    mov sp,#0xff00 ; 设置堆栈es:sp=0x9000:0xff00=0x9ff00
load_setup:
		mov dx,#0x0000          ; dh=磁头号; dl=驱动器号
    mov cx,#0x0002          ; ch=柱面号;cl=开始扇区
    mov bx,#0x0200          ; es:bx = 写入内存地址
    mov ax,#0x0200+SETUPLEN ; ah=0x02读操作; al=扇区数量
    int 0x13                ; 触发BIOS中断
    
    jnc ok_load_setup       ; 如果CF=0则跳转
    mov dx,#0x0000          
    mov ax,#0x0000          ; ah=0x02复位操作
    int 0x13
    j   load_setup          ; 重读
```

800

- 栈顶：`0x9ff00`
	- 栈的生长方向是由上往下的，即从高地址到低地址
	- 堆的生长方向是由下往上的，即从低地址到高地址

| 内存范围            | 内容      |
| --------------- | ------- |
| 0x90000-0x90100 | 启动程序    |
| 0x90200-91000   | setup模块 |
| 0x9ff00         | 栈顶      |

#### 加载系统模块

```
ok_load_setup:    
    ; 获取磁盘驱动器参数，特别是扇区/磁道的数量
    mov dl,#0x00        ; 类型为软盘
    mov ax,#0x0800      ; al=08读取驱动器参数
    int 0x13
    mov ch,#0x00
    seg cs
    mov sectors,cx      ; CL的位5-0＝扇区数
    mov ax,#INITSEG
    mov es,ax

    ; 打印无意义的信息
    
    mov ah,#0x03
    xor bh,bh
    int 0x10            ; 读光标位置
    
    mov cx,#24
    mov bx,#0x0007      ; 第0页，普通属性
    mov bp,#msg1
    mov ax,#1301        ; 写字符串，移动光标
    int 0x10            ; 显示字符

	; 加载 system 模块
    mov ax,#SYSSEG  
    mov es,ax
    call    read_it     ; 读入system模块
    jmpi    0,SETUPSEG  ; 转入0x9020:0x0000

read_it:    ; system模块可能很大，要跨越磁道
    mov ax,es
    cmp ax,#ENDSEG
    jb  ok1_read
    ret

ok1_read:
    mov ax,sectors
    sub ax,sread    ;sread时当前磁道已读扇区数，ax是未读扇区数
    call read_track ; 读磁道

; ... 省略若干可能


read_track:
	push ax
	push bx
	push cx
	push dx
	mov dx,track
	mov cx,sread
	inc cx
	mov ch,dl
	mov dx,head
	mov dh,dl
	mov dl,#0
	and dx,#0x0100
	mov ah,#2
	int 0x13
	jc bad_rt
	pop dx
	pop cx
	pop bx
	pop ax
	ret
bad_rt:	mov ax,#0
	mov dx,#0
	int 0x13
	pop dx
	pop cx
	pop bx
	pop ax
	jmp read_track
```


| 内存范围            | 内容       |
| --------------- | -------- |
| 0x10000         | system模块 |
| 0x90000-0x90100 | 启动程序     |
| 0x90200-91000   | setup模块  |
| 0x9ff00         | 栈顶       |
