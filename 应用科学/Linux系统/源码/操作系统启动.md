## 通过gdt/idt跳转地址

### 全局描述符表

![[计算机科学/操作系统/attachments/20250602220108.svg]]

全局描述符表(gdt)：在保护模式下，包含CS对应的内存基址
- S属性：存储了描述符的类型
	- S=0：系统段（System Segment）
	- S=1 数据段（Data Segment）或者代码段（Code Segment）
- TYPE属性：存储段的类型信息，随着S属性不同而不同
	- =1，TYPE\[3\]=0：数据段
		- TYPE\[0\]：段的访问标记，=0则该段未被访问
		- TYPE\[1\]：段的些权限，=0则只读
		- TYPE\[2\]：段的增长方向，=0则向上增长
	- =1，TYPE\[3\]=1：代码段
		- TYPE\[0\]：段的访问标记，=0则该段未被访问
		- TYPE\[1\]：段的读权限，=0则只能执行
		- TYPE\[2\]：一致代码段，=0则不是一致代码段
			- 一致代码段：用于权限是否要严格判定
	- =1：系统段
		- 0x1：可用286TSS
		- 0x2：该段存储了局部描述符表（LDT）
		- 0x3：忙的286TSS
		- 0x4：286调用门
		- 0x5：任务门
		- 0x6：286中断门
		- 0x7：286陷阱门
		- 0x9：可用386TSS
		- 0xB：忙的386TSS
		- 0xC：386调用门
		- 0xE：386中断门
		- 0xF：386陷阱门
- DPL属性：访问段的特权级，特权级范围为0～3，越小特权级越高
- P属性：标记了该段是否存在，=0则该段在内存中不存在
- AVL属性：由操作系统、应用程序自行定义
- L属性：仅在IA-32e模式下有意义，标记是否为64位代码段，=1则表示该段是64位
- G属性：记录了段界限的粒度
	- =0：段界限的粒度为字节
	- =1 时，段界限的粒度为4KB

中断描述符表(idt)：在保护模式下，包含int对应的中断处理函数入口


|位域|字段名|含义|
|---|---|---|
|0-15|`Offset Low`|处理程序地址的低 16 位。|
|16-31|`Selector`|代码段选择子（GDT/LDT 中的段描述符索引）。|
|32-34|`IST`|（可选）中断栈表索引（仅 TSS 中存在 IST 时有效）。|
|35-39|`Reserved`|保留位（必须为 0）。|
|40-43|`Type`|描述符类型（详见下表）。|
|44|`DPL`|描述符特权级（0=内核，3=用户态）。|
|45|`P`|存在位（1=有效，0=无效）。|
|46-63|`Offset High`|处理程序地址的高 16 位。|

### Type 字段取值

|值（二进制）|类型|用途|
|---|---|---|
|`1110` (0xE)|32-bit 中断门|常规中断，自动清除 IF 标志。|
|`1111` (0xF)|32-bit 陷阱门|异常/陷阱，不修改 IF 标志。|

### head子模块

```
startup_32:
	! 寄存器初始化
	movl $0x10,%eax
	mov %ax,%ds
	mov %ax,%es
	mov %ax,%fs
	mov %ax,%gs          ! 指向gdt的0x10项
	lss stack_start,%esp ! 栈指针初始化，从stack_start符号加载SS:ESP
	call setup_idt       ! 中断描述符表(IDT)初始化
	call setup_gdt       ! 全局描述符表(GDT)初始化
	
  ! 重新加载段寄存器
	movl $0x10,%eax	
	mov %ax,%ds		
	mov %ax,%es
	mov %ax,%fs
	mov %ax,%gs
	lss stack_start,%esp

	! 内存检测循环，当未开启A20时，寻址方式为实地址，地址0x10 0000和地址0x00 0000相同
	! 因为实地址只能解析20位地址，而地址0x10 0000是24位，超过部分会被忽略
	xorl %eax,%eax
1:incl %eax
	movl %eax,0x00 0000 ! 将1写入地址0x00 0000
	cmpl %eax,0x10 0000 ! 对比%eax和地址0x10 0000
	je   1b
	jmp after_page_tables


  ! 初始化中断描述符表(IDT)
setup_idt:
	lea ignore_int,%edx   ! edx = &ignore_int
	movl $0x00080000,%eax ! eax = 0008 0000
	movw %dx,%ax       ! ax = dx   (eax=0008 &ignore_int)
	movw $0x8E00,%dx   ! dx = 0x8E00(0000 8E00)
	
	lea idt,%edi       ! edi = &idt
	mov $256,%ecx      ! ecx = 256
rp_sidt:
	movl %eax,(%edi)   ! *edi= eax
	movl %edx,4(%edi)  ! (edi+4)=edx
	addl $8,%edi       ! edi += 8
	dec %ecx           ! --ecx
	jne rp_sidt        ! goto rp_sidt if cx!=0
	lidt idt_descr     
	ret

setup_gdt:
	lgdt gdt_descr
	ret

idt_descr:
	.word 256*8-1		
	.long idt
.align 2
.word 0
gdt_descr:
	.word 256*8-1		
	.long gdt	
	.align 8
idt:	.fill 256,8,0		
gdt:	.quad 0x0000000000000000
	.quad 0x00c09a0000000fff
	.quad 0x00c0920000000fff
	.quad 0x0000000000000000
	.fill 252,8,0		

```

## main.c

```C
void main(void)
{
  ROOT_DEV = ORIG_ROOT_DEV;
 	drive_info = DRIVE_INFO;
	memory_end = (1<<20) + (EXT_MEM_K<<10);
	memory_end &= 0xfffff000;
	if (memory_end > 16*1024*1024)
		memory_end = 16*1024*1024;
	if (memory_end > 12*1024*1024) 
		buffer_memory_end = 4*1024*1024;
	else if (memory_end > 6*1024*1024)
		buffer_memory_end = 2*1024*1024;
	else
		buffer_memory_end = 1*1024*1024;
	main_memory_start = buffer_memory_end;
#ifdef RAMDISK
	main_memory_start += rd_init(main_memory_start, RAMDISK*1024);
#endif
	mem_init(main_memory_start, memory_end);     // 初始化内存
	trap_init();    // 初始化硬盘
	blk_dev_init(); //
	chr_dev_init();
	tty_init();     // 初始化键盘
	time_init();    
	sched_init();
	buffer_init();
	hd_init();
	floppy_init();
	sti();
	mov_to_user_mode();
	if(!fock()) init();
	for(;;) pause();
}
```


```c
void mem_init(long start_mem, long end_mem)
{
	int i;
	HIGH_MEMORY = end_mem;
	for (i=0 ; i<PAGING_PAGES ; i++)
		mem_map[i] = USED;
	i = MAP_NR(start_mem);
	end_mem -= start_mem;
	end_mem >>= 12;
	while (end_mem-->0)
		mem_map[i++]=0;
}
```
