# 段页结合的实际内存管理

程序员希望用段、物理内存需要用页

## 段页结合
![[计算机科学/操作系统原理/attachments/2025-09-09_155955.svg]]

虚拟内存：操作系统开辟用于方便用于管理内存的虚拟地址空间，可以与物理地址进行映射

段页结合下的地址重定位
1. 通过段号和偏移(CS:IP)在虚拟内存表中查找虚拟基址
2. 虚拟基址通过页表查找物理页号
3. 物理页号和偏移控制物理内存

## 段页式内存管理

程序载入内存
- 分配建表：分配段和页并建立段表和页表
- 进程带动内存使用的图谱
- 从进程fork中的内存分配开始

1. 分割虚拟内存
2. 建立段表
3. 将段分割成页放置在物理内存
4. 建立页表
5. 重定位使用内存

### 内存管理实现

fork()=>sys_fork()=>copy_process();

kernal/system_call.s
```asm
sys_fork:  
    call find_empty_process  
    testl %eax,%eax  
    js 1f  
    push %gs  
    pushl %esi  
    pushl %edi  
    pushl %ebp  
    pushl %eax  
    call copy_process  
    addl $20,%esp  
1:  ret
```

```c
// 拷贝进程 nr 进程句柄
int copy_process(int nr,long ebp,long edi,long esi,long gs,long none,
		long ebx,long ecx,long edx,
		long fs,long es,long ds,
		long eip,long cs,long eflags,long esp,long ss)
{
	int i;
	struct file *f;

	struct task_struct *p = (struct task_struct *) get_free_page();
	
	if (!p)
		return -EAGAIN;
	task[nr] = p;
	*p = *current;	/* NOTE! this doesn't copy the supervisor stack */
	p->state = TASK_UNINTERRUPTIBLE;
	p->pid = last_pid;
	p->father = current->pid;
	p->counter = p->priority;
	p->signal = 0;
	p->alarm = 0;
	p->leader = 0;		/* process leadership doesn't inherit */
	p->utime = p->stime = 0;
	p->cutime = p->cstime = 0;
	p->start_time = jiffies;
	p->tss.back_link = 0;
	p->tss.esp0 = PAGE_SIZE + (long) p;
	p->tss.ss0 = 0x10;
	p->tss.eip = eip;
	p->tss.eflags = eflags;
	p->tss.eax = 0;
	p->tss.ecx = ecx;
	p->tss.edx = edx;
	p->tss.ebx = ebx;
	p->tss.esp = esp;
	p->tss.ebp = ebp;
	p->tss.esi = esi;
	p->tss.edi = edi;
	p->tss.es = es & 0xffff;
	p->tss.cs = cs & 0xffff;
	p->tss.ss = ss & 0xffff;
	p->tss.ds = ds & 0xffff;
	p->tss.fs = fs & 0xffff;
	p->tss.gs = gs & 0xffff;
	p->tss.ldt = _LDT(nr);
	p->tss.trace_bitmap = 0x80000000;
	if (last_task_used_math == current)
		__asm__("clts ; fnsave %0"::"m" (p->tss.i387));
	if (copy_mem(nr,p)) { // 
		task[nr] = NULL;
		free_page((long) p);
		return -EAGAIN;
	}
	for (i=0; i<NR_OPEN;i++)
		if ((f=p->filp[i]))
			f->f_count++;
	if (current->pwd)
		current->pwd->i_count++;
	if (current->root)
		current->root->i_count++;
	if (current->executable)
		current->executable->i_count++;
	set_tss_desc(gdt+(nr<<1)+FIRST_TSS_ENTRY,&(p->tss));// 设置tss任务描述符
	set_ldt_desc(gdt+(nr<<1)+FIRST_LDT_ENTRY,&(p->ldt));
	p->state = TASK_RUNNING;	/* do this last, just in case */
	return last_pid;
}
```

kernal/fork.c
```c
// 进程拷贝进内存
int copy_mem(int nr,struct task_struct * p) // nr 进程id
{
	unsigned long old_data_base,new_data_base,data_limit;// 数据段
	unsigned long old_code_base,new_code_base,code_limit;// 代码段

	code_limit=get_limit(0x0f);
	data_limit=get_limit(0x17);
	old_code_base = get_base(current->ldt[1]);//代码段 是当前ldt[1]的基地址
	old_data_base = get_base(current->ldt[2]);//数据段 是当前ldt[2]的基地址
	if (old_data_base != old_code_base)  // 要求数据段和代码段相同
		panic("We don't support separate I&D");
	if (data_limit < code_limit)
		panic("Bad data_limit");
	// 分配虚拟内存，建立段表
	new_data_base = new_code_base = nr * 0x4000000; // 分割虚拟地址
	p->start_code = new_code_base;
	set_base(p->ldt[1],new_code_base);// 保存代码段段表基址
	set_base(p->ldt[2],new_data_base);// 保存数据段段表基址
	
	//分配内存，建立页表
	if (copy_page_tables(old_data_base,new_data_base,data_limit)) { 
		printk("free_page_tables: from copy_mem\n");
		free_page_tables(new_data_base,data_limit);
		return -ENOMEM;
	}
	return 0;
}
```

```c
//拷贝页表
int copy_page_tables(unsigned long from,unsigned long to,long size)  
{  
    unsigned long * from_page_table;  
    unsigned long * to_page_table;  
    unsigned long this_page;  
    unsigned long * from_dir, * to_dir;  
    unsigned long nr;  
  
    if ((from&0x3fffff) || (to&0x3fffff))  
       panic("copy_page_tables called with wrong alignment");  
    from_dir = (unsigned long *) ((from>>20) & 0xffc);//父进程 页目录
    to_dir = (unsigned long *) ((to>>20) & 0xffc);    //子进程 页目录
    size = ((unsigned) (size+0x3fffff)) >> 22;  //拷贝大小
    
    for( ; size-- > 0 ; from_dir++,to_dir++) {  
       if (1 & *to_dir)  
          panic("copy_page_tables: already exist");  
       if (!(1 & *from_dir))  
          continue;
       from_page_table = (unsigned long *) (0xfffff000 & *from_dir); 
       // 分配一个页表 
       if (!(to_page_table = (unsigned long *) get_free_page())) 
          return -1; /* 内存溢出, see freeing */  
       *to_dir = ((unsigned long) to_page_table) | 7;  
       nr = (from==0)?0xA0:1024;
       // 复制所有页
       for ( ; nr-- > 0 ; from_page_table++,to_page_table++) {  
          this_page = *from_page_table;  
          if (!(1 & this_page))  
             continue;
          this_page &= ~2;  //设置为只读
          *to_page_table = this_page;  
          if (this_page > LOW_MEM) {  
             *from_page_table = this_page;  
             this_page -= LOW_MEM;  
             this_page >>= 12;
             mem_map[this_page]++;// 
          }  
       }  
    }  
    invalidate();  
    return 0;  
}
```
