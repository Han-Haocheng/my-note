### 含有中断的代码

```c
void main(){
	A();
	B();
}

void A(){
	fork();
}
```

对应代码段

```
! 100:0
main:
	call A
	call B
	ret

A:call fork
	ret
B: ret

fork:
	mov %eax,__NR_fork
	int 0x80           ! 100:123
	mov res,%eax       ! 100:125
user_stack:
	.dword 0,0,...,0
! 100:256
```

> [!NOTE] 注意：保护模式下 int 0x80中断会导致以下操作
> 1. 检查中断向量号有效性
> 2. 从IDT中获取中断描述符
> 3. 特权级检查：`CPL<DPL`
> 4. 切换栈
> 	- `mov ss,ax`
> 	- `mov esp,bx`
> 	- `mov IDT[ss],ss` 
> 	- `mov IDT[esp],esp` 
> 	- `push ax`
> 	- `push bx`
> 5. 保存上下文
> 	- `push EFLAGS`
> 	- `push CS`
> 	- `push EIP`
> 6. 跳转到目标代码

### 中断调用

kernel\system_call.s

```asm
kernel_stack:
	.dword 0,0     ! SS:ESP=100:256
	.dword 0       ! EFLAGS
	.dword 0,0     ! CS:EIP=100:125
	.dword 0,...,0 ! ds,...,fs,edx,...
	.dword 0

reschedule:
	pushl $ret_from_sys_call  # 保存返回函数
	jmp schedule              # 跳转
system_call:
	cmpl $nr_system_calls-1,%eax
	ja bad_sys_call
	push %ds
	push %es
	push %fs
	pushl %edx
	pushl %ecx		
	pushl %ebx		        # 保存 上下文
	movl $0x10,%edx		    # 将ds es段指向 0x10
	mov %dx,%ds           # 
	mov %dx,%es
	movl $0x17,%edx		    # 将fs指向0x17
	mov %dx,%fs
	
	call sys_call_table(,%eax,4) # 系统调用
	
	pushl $eax            # 保存返回值
	movl current,%eax
	cmpl $0,state(%eax)
	jne reschedule
	cmpl $0,counter(%eax)
	je reschedule         # 重调度条件 状态不是运行 或时间片为0
ret_from_sys_call:
	popl %eax
	popl %ebx...
	pop %fs...
	iret

sys_fork:
	push  %gs
	pushl %esi
	...
	push %eax
	call _copy_process
	addl %20,%esp
	ret
```

kernel\sched.c

```c
#include <linux\sched.h>

void schedule(void){
	int i,next,c;
	struct task_struct ** p;

/* check alarm, wake up any interruptible tasks that have got a signal */

	for(p = &LAST_TASK ; p > &FIRST_TASK ; --p)
		if (*p) {
			if ((*p)->alarm && (*p)->alarm < jiffies) {
					(*p)->signal |= (1<<(SIGALRM-1));
					(*p)->alarm = 0;
				}
			if (((*p)->signal & ~(_BLOCKABLE & (*p)->blocked)) &&
			(*p)->state==TASK_INTERRUPTIBLE)
				(*p)->state=TASK_RUNNING;
		}

/* this is the scheduler proper: */

	while (1) {
		c = -1;
		next = 0;
		i = NR_TASKS;
		p = &task[NR_TASKS];
		while (--i) {
			if (!*--p)
				continue;
			if ((*p)->state == TASK_RUNNING && (*p)->counter > c)
				c = (*p)->counter, next = i;
		}
		if (c) break;
		for(p = &LAST_TASK ; p > &FIRST_TASK ; --p)
			if (*p)
				(*p)->counter = ((*p)->counter >> 1) +
						(*p)->priority;
	}
	switch_to(next);
}
```

### 复制进程为子进程

```c
int copy_process(int nr,
	long ebp,long edi,long esi,long gs,long none,
	long ebx,long ecx,long edx,
	long fs,long es,long ds,
	long eip,long cs,long eflags,long esp,long ss){
	//申请内存
	p=(struct task_struct *)get_free_page();
	// 创建内核栈
	p->tss.ss0 =0x10;
	p->tss.esp0=PAGE_SIZE + (long)p;
	//创建用户栈
	p->tss.ss  = ss&0xffff;
	p->tss.esp = esp;
	p->tss.eip = eip
	p->tss.cs  = cs&0xffff;
	p->tss.eax = 0
	p->tss.ecx = ecx;
	p->tss.ldt = _LDT(NR);
	set_tss_desc(gdt+(nr<<1))+FIRST_TSS_ENTRY,&(p->tss))
	set_idt_desc(gdt+(nr<<1))+FIRST_IDT_ENTRY,&(p->idt))
	p->state=TASK_RUNNINT;
}
```
