# 进程同步


## 生产者-消费者问题


```c
// P0
while(true){
	if(counter==BUFFER_SIZE)
		sleep();
	buffer[in]=item;
	in=(in+1)%BUFFER_SIZE;
	counter++;
	if(counter==1)wakeup(P1)
}

//P1
while(true){
	if(counter==0)
		sleep();
	item = buffer[out];
	out=(out+1)%BUFFER_SIZE;
	counter--;
	if(counter==BUFFER_SIZE-1)wakeup(P0)
}
```

问题：无法处理多个生产者或多个消费者被唤醒的问题
解决：引入信号量，每一次睡眠都记录一次，每一次唤醒都减少一次记录

### 信号量实现
```c
struct semaphore{
	int value;
	PCB *queue;
}

void Proberen(semaphore s){
	s.value--;
	if(s.value<0){
		sleep(s,queue);
	}
}

void Verhogen(semaphore s){
	s.value++;
	if(s.value<=0){
		wakeup(s.queue);
	}
}
```

### 解决生产者-消费者问题


```c
int fd = open("buffer.text");

write(fd,0,sizeof(int)); //写in
write(fd,0,sizeof(int)); //写out

semaphore full = 0;
semaphore empty = BUFFER_SIZE;
semaphore mutex = 1;

void Producer(item){
	P(empty);
	
	loc = setloc(fd);
	P(mutex);
	write(item,loc);
	V(mutex);
	
	V(full);
}

void Consumer(item){
	P(empty)
	
	loc = getloc(fd);
	P(mutex)
	item = read(loc);
	V(mutex);
	printf(item);
	
	V(full);
}
```
